import matplotlib.pyplot as plt
import numpy as np
import pandas as pd
from datetime import datetime, timedelta
import seaborn as sns
import warnings
warnings.filterwarnings('ignore')

class AgricultureClimateAnalyzer:
    def __init__(self):
        self.crop_types = ['Wheat', 'Rice', 'Maize', 'Soybean', 'Potato', 'Vegetables', 'Fruits']
        self.weather_events = ['Drought', 'Floods', 'Heatwaves', 'Frost', 'Storms', 'Wildfires']
        self.regions = ['North America', 'Europe', 'Asia', 'Africa', 'South America', 'Oceania']
        
    def analyze_crop_vulnerability(self):
        """Analyze vulnerability of different crops to extreme weather"""
        print("=== CROP VULNERABILITY ANALYSIS ===")
        
        # Vulnerability scores (0-10 scale) for each crop-weather combination
        vulnerability_data = {
            'Drought': [8, 9, 7, 6, 5, 8, 7],
            'Floods': [6, 9, 5, 4, 8, 9, 6],
            'Heatwaves': [7, 8, 6, 5, 4, 7, 6],
            'Frost': [6, 3, 4, 2, 8, 9, 8],
            'Storms': [5, 6, 5, 4, 6, 7, 8],
            'Wildfires': [4, 3, 3, 2, 2, 5, 6]
        }
        
        vulnerability_df = pd.DataFrame(vulnerability_data, index=self.crop_types)
        
        print("\nCrop Vulnerability to Extreme Weather (0-10 scale):")
        print(vulnerability_df.round(2))
        
        return vulnerability_df
    
    def yield_reduction_analysis(self):
        """Analyze projected yield reductions due to climate extremes"""
        print("\n=== CROP YIELD REDUCTION ANALYSIS ===")
        
        scenarios = ['Current', '+1C', '+2C', '+3C', '+4C']
        
        yield_data = {
            'Wheat': [0, -5, -12, -25, -40],
            'Rice': [0, -8, -18, -32, -50],
            'Maize': [0, -10, -22, -38, -55],
            'Soybean': [0, -6, -15, -28, -45],
            'Potato': [0, -7, -16, -30, -48],
            'Vegetables': [0, -12, -25, -42, -60],
            'Fruits': [0, -15, -30, -50, -65]
        }
        
        yield_df = pd.DataFrame(yield_data, index=scenarios)
        
        print("Projected Yield Reduction (% change from baseline):")
        print(yield_df.round(1))
        
        return yield_df
    
    def regional_impact_assessment(self):
        """Assess regional impacts on food production"""
        print("\n=== REGIONAL IMPACT ASSESSMENT ===")
        
        # Impact scores (0-10 scale) for each region-weather combination
        regional_impacts = {
            'Drought': [6, 4, 8, 9, 5, 7],
            'Floods': [5, 6, 9, 7, 8, 4],
            'Heatwaves': [7, 6, 8, 9, 6, 5],
            'Frost': [8, 7, 4, 2, 3, 4],
            'Storms': [7, 5, 9, 6, 8, 6],
            'Wildfires': [8, 5, 4, 3, 6, 9]
        }
        
        regional_df = pd.DataFrame(regional_impacts, index=self.regions)
        
        print("Regional Vulnerability to Climate Extremes (0-10 scale):")
        print(regional_df.round(2))
        
        return regional_df
    
    def food_security_analysis(self):
        """Analyze impacts on food security metrics"""
        print("\n=== FOOD SECURITY IMPACT ANALYSIS ===")
        
        years = np.array([2024, 2030, 2040, 2050])
        
        security_metrics = {
            'Year': years,
            'Global_Food_Production_Index': [100, 98, 92, 85],
            'Food_Price_Index': [100, 115, 145, 190],
            'Undernourished_Population_Millions': [690, 720, 780, 850],
            'Crop_Failure_Frequency': [100, 120, 150, 190],
            'Food_Import_Dependency_Index': [100, 110, 125, 145]
        }
        
        security_df = pd.DataFrame(security_metrics)
        
        print("Food Security Projections (Index: 2024=100):")
        print(security_df.round(1))
        
        return security_df
    
    def economic_impact_assessment(self):
        """Assess economic impacts on agriculture sector"""
        print("\n=== ECONOMIC IMPACT ASSESSMENT ===")
        
        impact_categories = [
            'Crop Losses', 'Livestock Losses', 'Infrastructure Damage',
            'Increased Irrigation Costs', 'Pest Control Costs', 'Insurance Premiums'
        ]
        
        economic_data = {
            'Current_Cost_Billion_USD': [45, 28, 15, 12, 8, 5],
            'Projected_2030_Billion_USD': [65, 40, 22, 18, 12, 8],
            'Projected_2050_Billion_USD': [120, 75, 45, 35, 25, 15]
        }
        
        economic_df = pd.DataFrame(economic_data, index=impact_categories)
        
        print("Economic Impacts of Climate Extremes (Annual Costs in Billion USD):")
        print(economic_df.round(1))
        
        return economic_df
    
    def adaptation_strategies(self):
        """Recommend adaptation strategies for agriculture"""
        print("\n=== ADAPTATION STRATEGIES ===")
        
        strategies = {
            'Crop Management': [
                "Drought-resistant crop varieties",
                "Modified planting dates and seasons",
                "Crop diversification and rotation",
                "Conservation agriculture practices"
            ],
            'Water Management': [
                "Improved irrigation efficiency",
                "Water harvesting and storage",
                "Drip irrigation systems",
                "Soil moisture conservation"
            ],
            'Technology Solutions': [
                "Climate-smart agriculture technologies",
                "Early warning systems for extreme weather",
                "Precision farming techniques",
                "Weather-based crop insurance"
            ],
            'Policy Interventions': [
                "Climate-resilient food security policies",
                "Support for smallholder farmers",
                "Research and development funding",
                "International cooperation on food security"
            ]
        }
        
        for category, strategy_list in strategies.items():
            print(f"\n{category}:")
            for i, strategy in enumerate(strategy_list, 1):
                print(f"  {i}. {strategy}")
        
        return strategies
    
    def climate_resilient_crops(self):
        """Analyze potential of climate-resilient crop varieties"""
        print("\n=== CLIMATE-RESILIENT CROP ANALYSIS ===")
        
        resilient_crops = {
            'Crop_Type': self.crop_types,
            'Drought_Tolerance_Improvement': [35, 40, 30, 25, 20, 15, 25],
            'Heat_Tolerance_Improvement': [30, 35, 40, 20, 25, 30, 35],
            'Flood_Tolerance_Improvement': [20, 25, 15, 10, 30, 25, 20],
            'Development_Status': ['Advanced', 'Intermediate', 'Advanced', 'Early', 'Intermediate', 'Advanced', 'Intermediate']
        }
        
        resilient_df = pd.DataFrame(resilient_crops)
        
        print("Climate-Resilient Crop Varieties (% Improvement over Conventional):")
        print(resilient_df.to_string(index=False))
        
        return resilient_df
    
    def create_visualizations(self, vulnerability_df, yield_df, regional_df, security_df, economic_df, resilient_df):
        """Create comprehensive visualizations"""
        plt.style.use('seaborn-v0_8')
        fig, axes = plt.subplots(3, 2, figsize=(18, 16))
        fig.suptitle('Extreme Weather Impact on Agriculture and Food Security', fontsize=16, fontweight='bold')
        
        # 1. Crop Vulnerability Heatmap
        sns.heatmap(vulnerability_df, annot=True, cmap='Reds', ax=axes[0, 0])
        axes[0, 0].set_title('Crop Vulnerability to Extreme Weather Events')
        axes[0, 0].tick_params(axis='x', rotation=45)
        axes[0, 0].tick_params(axis='y', rotation=0)
        
        # 2. Yield Reduction Projections
        years = [0, 1, 2, 3, 4]  # Temperature increase
        for crop in yield_df.columns:
            axes[0, 1].plot(years, yield_df[crop], marker='o', label=crop, linewidth=2)
        axes[0, 1].set_xlabel('Temperature Increase (C)')
        axes[0, 1].set_ylabel('Yield Reduction (%)')
        axes[0, 1].set_title('Crop Yield Reduction Projections')
        axes[0, 1].legend(bbox_to_anchor=(1.05, 1), loc='upper left')
        axes[0, 1].grid(True, alpha=0.3)
        axes[0, 1].set_xticks(years)
        
        # 3. Regional Impact Heatmap
        sns.heatmap(regional_df, annot=True, cmap='YlOrRd', ax=axes[1, 0])
        axes[1, 0].set_title('Regional Vulnerability to Climate Extremes')
        axes[1, 0].tick_params(axis='x', rotation=45)
        axes[1, 0].tick_params(axis='y', rotation=0)
        
        # 4. Food Security Trends
        security_metrics = ['Global_Food_Production_Index', 'Food_Price_Index', 
                          'Crop_Failure_Frequency', 'Food_Import_Dependency_Index']
        security_df.set_index('Year')[security_metrics].plot(ax=axes[1, 1], marker='o', linewidth=2)
        axes[1, 1].set_xlabel('Year')
        axes[1, 1].set_ylabel('Index (2024=100)')
        axes[1, 1].set_title('Food Security Metrics Projection')
        axes[1, 1].legend(bbox_to_anchor=(1.05, 1), loc='upper left')
        axes[1, 1].grid(True, alpha=0.3)
        
        # 5. Economic Impacts
        x_pos = np.arange(len(economic_df.index))
        width = 0.25
        
        axes[2, 0].bar(x_pos - width, economic_df['Current_Cost_Billion_USD'], width, 
                      label='Current', alpha=0.8)
        axes[2, 0].bar(x_pos, economic_df['Projected_2030_Billion_USD'], width, 
                      label='2030 Projected', alpha=0.8)
        axes[2, 0].bar(x_pos + width, economic_df['Projected_2050_Billion_USD'], width, 
                      label='2050 Projected', alpha=0.8)
        
        axes[2, 0].set_xlabel('Impact Category')
        axes[2, 0].set_ylabel('Cost (Billion USD)')
        axes[2, 0].set_title('Economic Impacts of Climate Extremes')
        axes[2, 0].set_xticks(x_pos)
        axes[2, 0].set_xticklabels(economic_df.index, rotation=45, ha='right')
        axes[2, 0].legend()
        axes[2, 0].grid(True, alpha=0.3)
        
        # 6. Resilient Crop Improvements
        resilient_metrics = ['Drought_Tolerance_Improvement', 'Heat_Tolerance_Improvement', 
                           'Flood_Tolerance_Improvement']
        x_pos = np.arange(len(resilient_df['Crop_Type']))
        width = 0.25
        
        for i, metric in enumerate(resilient_metrics):
            axes[2, 1].bar(x_pos + i*width, resilient_df[metric], width, label=metric.replace('_', ' '))
        
        axes[2, 1].set_xlabel('Crop Type')
        axes[2, 1].set_ylabel('Improvement (%)')
        axes[2, 1].set_title('Climate-Resilient Crop Improvements')
        axes[2, 1].set_xticks(x_pos + width)
        axes[2, 1].set_xticklabels(resilient_df['Crop_Type'], rotation=45, ha='right')
        axes[2, 1].legend(bbox_to_anchor=(1.05, 1), loc='upper left')
        axes[2, 1].grid(True, alpha=0.3)
        
        plt.tight_layout()
        plt.savefig('agriculture_climate_impact_analysis.png', dpi=300, bbox_inches='tight')
        plt.show()
    
    def risk_assessment_matrix(self):
        """Create risk assessment matrix for different scenarios"""
        print("\n=== RISK ASSESSMENT MATRIX ===")
        
        scenarios = [
            "Moderate Warming (+1.5C)",
            "High Warming (+2.5C)", 
            "Very High Warming (+4C)",
            "Frequent Extreme Events",
            "Compound Events"
        ]
        
        risk_levels = ['Low', 'Medium', 'High', 'Very High', 'Critical']
        
        risk_data = {
            'Food_Production_Risk': [1, 2, 4, 3, 4],
            'Price_Volatility_Risk': [1, 2, 4, 3, 4],
            'Supply_Chain_Risk': [1, 2, 3, 3, 4],
            'Smallholder_Vulnerability': [2, 3, 4, 3, 4],
            'Global_Security_Risk': [1, 2, 3, 2, 3]
        }
        
        risk_df = pd.DataFrame(risk_data, index=scenarios)
        
        print("Risk Assessment Matrix (1=Low, 5=Critical):")
        print(risk_df)
        
        return risk_df
    
    def generate_report(self):
        """Generate comprehensive analysis report"""
        print("="*70)
        print("EXTREME WEATHER IMPACT ON AGRICULTURE AND FOOD SECURITY")
        print("="*70)
        print(f"Generated on: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        print("\n")
        
        # Run all analyses
        vulnerability_df = self.analyze_crop_vulnerability()
        yield_df = self.yield_reduction_analysis()
        regional_df = self.regional_impact_assessment()
        security_df = self.food_security_analysis()
        economic_df = self.economic_impact_assessment()
        resilient_df = self.climate_resilient_crops()
        strategies = self.adaptation_strategies()
        risk_df = self.risk_assessment_matrix()
        
        # Create visualizations
        self.create_visualizations(vulnerability_df, yield_df, regional_df, security_df, economic_df, resilient_df)
        
        # Summary conclusions
        print("\n" + "="*70)
        print("KEY FINDINGS AND URGENT ACTIONS")
        print("="*70)
        
        key_findings = [
            " Rice and vegetables show highest vulnerability to multiple climate extremes",
            " Projected yield reductions could reach 50-65% for key crops by 2050",
            " Africa and Asia face the highest regional vulnerability scores",
            " Food prices could nearly double by 2050 without adaptation measures",
            " Economic costs may exceed $300 billion annually by 2050",
            " Climate-resilient crop varieties offer 20-40% improvement potential",
            " Compound climate events pose the highest systemic risks"
        ]
        
        for finding in key_findings:
            print(finding)
        
        urgent_actions = [
            "\nURGENT ACTIONS REQUIRED:",
            "1. Accelerate development and deployment of climate-resilient crops",
            "2. Invest in water-efficient irrigation and storage infrastructure", 
            "3. Strengthen early warning systems for extreme weather events",
            "4. Support smallholder farmers with climate adaptation resources",
            "5. Diversify food production systems and supply chains",
            "6. Implement climate-smart agricultural policies globally"
        ]
        
        for action in urgent_actions:
            print(action)
        
        print(f"\nReport generated successfully. Visualization saved as 'agriculture_climate_impact_analysis.png'")

def main():
    """Main function to run the agriculture climate impact analysis"""
    analyzer = AgricultureClimateAnalyzer()
    analyzer.generate_report()

if __name__ == "__main__":
    main()
